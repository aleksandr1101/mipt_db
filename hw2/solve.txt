docker pull mongo
docker run --name my_mongodb -d -p 27017:27017 mongo
docker exec -it my_mongodb sh
добавляем датасет https://www.kaggle.com/c/fake-news/data?select=train.csv через mongoinsert в коллекцию my_collection
docker exec -it my_mongodb mongosh
